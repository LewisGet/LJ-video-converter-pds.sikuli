def create(value, s, e, layout, color, size):
    setStart(s)
    setLayout(layout)
    createBlock(color)
    setText(value, size)
    setEnd(e)

def createBlock(color):
    if (color == "blue"):
        rightClick(Pattern().targetOffset(173,-97))
    else:
        rightClick(Pattern().targetOffset(25,-97))

    click(Pattern().targetOffset(-51,-93))
    wait()

def setStart(s):
    click(Pattern().targetOffset(-42,-15))
    doubleClick(Pattern().targetOffset(41,-14))
    type(s[0])
    type(s[1])
    type(s[2])
    type(Key.ENTER)

def setEnd(e):
    wait(Pattern().targetOffset(-31,1))
    click(Pattern().targetOffset(100,1))
    doubleClick(Pattern().targetOffset(-10,-14))
    type(e[0])
    type(e[1])
    type(e[2])
    click(Pattern().targetOffset(2,33))

def setText(value, font):
    click(Pattern().targetOffset(-31,1))
    wait()
    doubleClick(Pattern().targetOffset(-65,-26))
    type("a", Key.CTRL)
    paste(value.decode("utf8"))
    setFontSize(font)
    click(Pattern().targetOffset(77,13))

def setLayout(value):
    if (value == 1):
        click(Pattern().targetOffset(-14,-209))
    if (value == 2):
        click(Pattern().targetOffset(-11,-171))
    if (value == 3):
        click(Pattern().targetOffset(-14,-134))
    if (value == 4):
        click(Pattern().targetOffset(-14,-100))
    if (value == 5):
        click(Pattern().targetOffset(-12,-61))
    if (value == 6):
        click(Pattern().targetOffset(-14,-23))
    if (value == 7):
        click(Pattern().targetOffset(-13,12))
    if (value == 8):
        click(Pattern().targetOffset(-12,50))

def setFontSize(value):
    doubleClick(Pattern().targetOffset(-14,-29))
    type(value)
